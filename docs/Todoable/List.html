def items
  output = @todoable.request!("get", LIST_PATH, { :list_id => @id })
  output['items'].collect { |json_item|
    Item.new(@todoable, self, json_item)
  }
end

def update name
  body = {"list": {"name": name}}.to_json
  @todoable.request!("patch", LIST_PATH, { :list_id => @id }, body)
  @name = name
end
